对R中的数据集faithful, 分别使用乘积和和球对称核估计eruptions和waiting的联合密度函数, 考虑不同带宽选择方法.

library('scatterplot3d')

fai=as.matrix(faithful)
# default bandwidth
f=KernSur(fai[,1],fai[,2])
x=rep(f$xords,length(f$xords))
y=kronecker(f$yords,rep(1,length(f$yords)))
z=as.vector(f$zden)
scatterplot3d(x,y,z,type = 'l')

# plug-in bandwidth 
# Scott's bandwidth 
H=sqrt(cov(fai))*nrow(fai)^{-1/6}
h1=H[1,1]
h2=H[2,2]
f=KernSur(fai[,1],fai[,2],xbandwidth = h1,ybandwidth = h2)
x=rep(f$xords,length(f$xords))
y=kronecker(f$yords,rep(1,length(f$yords)))
z=as.vector(f$zden)
scatterplot3d(x,y,z,type = 'l')

# Silverman's bandwidth 
H=sqrt(cov(fai))*(3*nrow(fai)/4)^{-1/5}
h1=H[1,1]
h2=H[2,2]
f=KernSur(fai[,1],fai[,2],xbandwidth = h1,ybandwidth = h2)
x=rep(f$xords,length(f$xords))
y=kronecker(f$yords,rep(1,length(f$yords)))
z=as.vector(f$zden)
scatterplot3d(x,y,z,type = 'l')
