Problem

Generate a dataset, use the function solved in assignment 2 to estimate, and use cross-validation to choose the optimal bandwidth.

estimation

# Initialization
n <- 200 
truebeta <- c(2,2) #true beta_0 and beta_1
lambda <- function(x) exp(truebeta[1]+truebeta[2]*x)
lik <- function(y,beta) beta-exp(beta)*y #likelihood function with $\lambda=e^\hat{\beta}$
set.seed(0)
# Generate a dataset
X <- runif(n = n, -3, 3)
Y <- rexp(n = n, rate = lambda(X))

# Set bandwidth and evaluation grid
h <- 0.25
x <- seq(-3, 3, l = 501)

# Optimize the weighted log-likelihood explicitly
suppressWarnings(
  fitNlm <- sapply(x, function(x) {
    K <- dnorm(x = x, mean = X, sd = h)
    nlm(f = function(beta) {
      sum(K * (Y * exp(beta[1] + beta[2] * (X - x)) - ( beta[1] + beta[2] * (X - x))))
    }, p = c(0, 0))$estimate[1]
  })
)

# Compare fits
plot(x, 1/lambda(x),  type = "l", lwd = 2)
lines(x,exp(-fitNlm), col = 2, lty = 2) #inverse of link function g

cross-validation

We conclude that the optimal bandwidth is h=0.65.

# Exact LCV 
h <- seq(0.1,1, by = .05)
suppressWarnings(
  LCV <- sapply(h, function(h) {
    sum(sapply(1:n, function(i) {
      K <- dnorm(x = X[i], mean = X[-i], sd = h)
      lik(Y[i],nlm(f = function(beta) {
        sum(K * (Y[-i] * exp(beta[1] + beta[2] * (X[-i] - X[i])) - ( beta[1] + beta[2] * (X[-i] - X[i]))))
      }, p = c(0, 0))$estimate[1])
    }))
  })
)
plot(h, LCV, type = "o")
abline(v = h[which.max(LCV)], col = 2)
